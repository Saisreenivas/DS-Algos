/**
 * random array -> positive and negative integers
 * output should be sorted array
 * input = 2,-4,1,-3,5
 * output = -4,-3,1,2,5
 */

const main = () => {
    const input = [2, -4, 1, -3, 5];
    for (let i = 1; i < input.length; i++) {
        compareAndPlaceElement(i, input);
    }
    console.log(input);
}

const compareAndPlaceElement = (itemIndex, inputArray) =>{
    for(let i=0;i<itemIndex;i++){
        if(inputArray[i]> inputArray[itemIndex]){
            shiftData(inputArray, i, itemIndex);
            console.log(inputArray)
            return;
        }
    }
    // const index = Math.floor(itemIndex/2);

}

const shiftData = (inputArray, fromindex, toIndex) =>{
    console.log(inputArray)
    const tmp = inputArray[fromindex];
    for(let i=(fromindex+1);i<= toIndex;i++){
        inputArray[i-1] = inputArray[i];
    }
    inputArray[toIndex] = tmp;
    console.log(inputArray)
}


main();
